caption = "Fig 9: Breakdown, by artist, of all the songs listened to in <br> Week 1 (March 9, 2019 - March 15, 2019). Artists that make up at least 50% of <br> weekly listening are pulled out." 
time_period_data = a1.get_time_periods_dfs(data)
for d in time_period_data.values():
    props = d['props']
    props['Artist'] = props.index
    props = props.sort_values(by = 'proportion', ascending = False)
    half_prop = 0
    pull_arr = np.zeros(len(props.index))
    count_pull = 0
    while half_prop < 0.5:
        p = props.iloc[count_pull]['proportion']
        half_prop += p
        pull_arr[count_pull] = 0.25
        count_pull += 1
    fig = go.Figure(go.Pie(labels=props['Artist'], values=props['count'], pull = pull_arr))
    fig.update_layout(uniformtext_minsize=14, uniformtext_mode='hide')
    fig.update_traces(textposition='inside', textinfo='percent+label',
                          textfont_size=20, marker=dict(line=dict(color='#000000', width=1)))
    fig.update_layout(title_text="Songs Listened to in Week 1: Divided by Artist")
